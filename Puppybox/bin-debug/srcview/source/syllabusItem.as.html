package
{
    public class syllabusItem
    {
        public var difficulty:String;
        public var item1:String;
        public var item2:String;        
        
        public function syllabusItem(sItem:XML)
        {
            difficulty = sItem.difficulty;
            item1 = sItem.item1;
            item2 = sItem.item2;
        }
    }
}
